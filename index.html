function doGet(e) { const ss = SpreadsheetApp.getActiveSpreadsheet(); // ── proxy לכנסת ── if (e.parameter.voteId) { const url = "https://www.knesset.gov.il/WebSiteApi/knessetapi/Votes/GetVoteDetails/" + e.parameter.voteId; const response = UrlFetchApp.fetch(url); return ContentService .createTextOutput(response.getContentText()) .setMimeType(ContentService.MimeType.JSON); } // ── תוצאות משחק זיכרון ── if (e.parameter.scores === "1") { const difficulty = e.parameter.difficulty || "medium"; let sheet = ss.getSheetByName("תוצאות"); if (!sheet) return json([]); const rows = sheet.getDataRange().getValues(); const data = rows.slice(1) .filter(r => r[1] === difficulty) .map(r => ({ timestamp: r[0], difficulty: r[1], pairs: r[2], moves: r[3], time_seconds: r[4], accuracy: r[5], vote_id: r[6], vote_short_name: r[7], vote_type: r[8], })) .sort((a, b) => a.moves !== b.moves ? a.moves - b.moves : a.time_seconds - b.time_seconds) .slice(0, 20); return json(data); } // ── לידרבורד משחק פיצוץ ── if (e.parameter.blaster === "1") { const difficulty = e.parameter.difficulty || "medium"; let sheet = ss.getSheetByName("תוצאות_משחק_פיצוץ"); if (!sheet) return json([]); const rows = sheet.getDataRange().getValues(); const data = rows.slice(1) .filter(r => r[1] === difficulty) .map(r => ({ timestamp: r[0], difficulty: r[1], hits: r[2], misses: r[3], accuracy: r[4], vote_id: r[5], vote_short_name: r[6], vote_type: r[7], })) .sort((a, b) => b.hits !== a.hits ? b.hits - a.hits : b.accuracy - a.accuracy) .slice(0, 20); return json(data); } // ── לידרבורד משחק סווייפ ── if (e.parameter.swipe === "1") { const difficulty = e.parameter.difficulty || "medium"; const voteId = e.parameter.vote_id || ""; let sheet = ss.getSheetByName("swipe_kneset"); if (!sheet) return json([]); const rows = sheet.getDataRange().getValues(); const data = rows.slice(1) .filter(r => r[1] === difficulty && (!voteId || String(r[5]) === voteId)) .map(r => ({ timestamp: r[0], difficulty: r[1], correct: r[2], wrong: r[3], accuracy: r[4], vote_id: r[5], vote_short_name: r[6], user_vote: r[7], })) .sort((a, b) => b.accuracy !== a.accuracy ? b.accuracy - a.accuracy : b.correct - a.correct) .slice(0, 20); return json(data); } // ── רשימת הצבעות (פעילות בלבד) ── const sheet = ss.getSheets()[0]; const rows = sheet.getDataRange().getValues(); const data = rows .filter(r => String(r[0]).match(/^\d+$/)) .filter(r => String(r[3]).trim() === "כן") .map(r => ({ voteId: String(r[0]).trim(), type: String(r[1]).trim(), shortName: String(r[2] || "").trim(), })); return json(data); } function doPost(e) { try { const payload = JSON.parse(e.postData.contents); const ss = SpreadsheetApp.getActiveSpreadsheet(); const tz = Session.getScriptTimeZone(); const now = Utilities.formatDate(new Date(), tz, "yyyy-MM-dd HH:mm:ss"); // ── משחק סווייפ ── if (payload.game === "swipe") { let sheet = ss.getSheetByName("swipe_kneset"); if (!sheet) { sheet = ss.insertSheet("swipe_kneset"); sheet.appendRow([ "timestamp", "difficulty", "correct", "wrong", "accuracy", "vote_id", "vote_short_name", "user_vote" ]); } sheet.appendRow([ now, payload.difficulty || "", payload.correct || 0, payload.wrong || 0, payload.accuracy || 0, payload.vote_id || "", payload.vote_short_name || "", payload.user_vote || "", ]); return json({ ok: true }); } // ── משחק פיצוץ ── if (payload.game === "blaster") { let sheet = ss.getSheetByName("תוצאות_משחק_פיצוץ"); if (!sheet) { sheet = ss.insertSheet("תוצאות_משחק_פיצוץ"); sheet.appendRow([ "timestamp", "difficulty", "hits", "misses", "accuracy", "vote_id", "vote_short_name", "vote_type" ]); } sheet.appendRow([ now, payload.difficulty || "", payload.hits || 0, payload.misses || 0, payload.accuracy || 0, payload.vote_id || "", payload.vote_short_name || "", payload.vote_type || "", ]); return json({ ok: true }); } // ── משחק זיכרון ── let sheet = ss.getSheetByName("תוצאות"); if (!sheet) { sheet = ss.insertSheet("תוצאות"); sheet.appendRow([ "timestamp", "difficulty", "pairs", "moves", "time_seconds", "accuracy", "vote_id", "vote_short_name", "vote_type" ]); } sheet.appendRow([ now, payload.difficulty || "", payload.pairs || "", payload.moves || "", payload.time_seconds || "", payload.accuracy || "", payload.vote_id || "", payload.vote_short_name || "", payload.vote_type || "", ]); return json({ ok: true }); } catch(err) { return json({ ok: false, error: err.message }); } } function json(data) { return ContentService .createTextOutput(JSON.stringify(data)) .setMimeType(ContentService.MimeType.JSON); }
